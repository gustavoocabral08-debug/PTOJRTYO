    async function loadVehiclesFromGitHub() {
        try {
            isLoading = true;
            render();
            
            const response = await fetch(GITHUB_API_URL, {
                headers: {
                    'Authorization': `token ${GITHUB_TOKEN}`,
                    'Accept': 'application/vnd.github.v3+json'
                }
            });
            
            if (response.ok) {
                const data = await response.json();
                currentSha = data.sha;
                const content = atob(data.content);
                vehicles = JSON.parse(content);
                showToast('Dados carregados do GitHub!');
            } else {
                console.error('Erro ao carregar:', response.status);
                showToast('Erro ao carregar dados. Usando dados locais.', true);
                loadLocalData();
            }
        } catch (error) {
            console.error('Erro:', error);
            showToast('Erro de conexão. Usando dados locais.', true);
            loadLocalData();
        } finally {
            isLoading = false;
            render();
        }
    }

    async function saveVehiclesToGitHub() {
        if (!isAdmin) return;
        
        try {
            const content = btoa(unescape(encodeURIComponent(JSON.stringify(vehicles, null, 2))));
            
            const response = await fetch(GITHUB_API_URL, {
                method: 'PUT',
                headers: {
                    'Authorization': `token ${GITHUB_TOKEN}`,
                    'Accept': 'application/vnd.github.v3+json',
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({
                    message: 'Atualização de veículos via site',
                    content: content,
                    sha: currentSha
                })
            });
            
            if (response.ok) {
                const data = await response.json();
                currentSha = data.content.sha;
                showToast('Dados salvos no GitHub com sucesso!');
                return true;
            } else {
                showToast('Erro ao salvar no GitHub', true);
                return false;
            }
        } catch (error) {
            console.error('Erro ao salvar:', error);
            showToast('Erro de conexão ao salvar', true);
            return false;
        }
    }
